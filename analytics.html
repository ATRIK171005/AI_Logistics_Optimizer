SELECT 
    hub.facility_code,
    hub.region_name,
    COUNT(opt.run_id) AS total_optimizations,
    ROUND(AVG(opt.cost_savings_pct), 2) AS avg_savings_percentage,
    SUM(opt.distance_saved_km) AS cumulative_km_saved
FROM logistics_hubs hub
JOIN optimization_history opt ON hub.hub_id = opt.hub_id
WHERE opt.created_at >= '2024-01-01'
GROUP BY hub.facility_code, hub.region_name
ORDER BY avg_savings_percentage DESC;
